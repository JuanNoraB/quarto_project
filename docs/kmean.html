sse = [] #inertia
k_range = range(2,11) #valores posibles de k

for k in k_range:
    kmeans = KMeans(n_clusters=k,random_state=42)
    kmeans.fit(df[['age','annual_income($)']])
    sse.append(kmeans.inertia_)

print("Inertia:\n",sse)
